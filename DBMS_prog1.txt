CREATE TABLE PUBLISHER0033(
Name VARCHAR(10) PRIMARY KEY,
Address VARCHAR(20),
phone INT);

CREATE TABLE BOOK0033(
Book_id INT PRIMARY KEY,
Title VARCHAR(20),
Publisher_name VARCHAR(20) REFERENCES PUBLISHER0033(Name)
ON DELETE CASCADE,
Pub_Year DATE
);

CREATE TABLE BOOKAUTHORS0033(
Book_id INT REFERENCES BOOK0033(Book_id) ON DELETE CASCADE,
Author_name VARCHAR(20), 
PRIMARY KEY(Author_name,Book_id)
);
CREATE TABLE LIBRARYBRANCH0033(
Branch_id INT PRIMARY KEY,
Branch_name VARCHAR(20),
Address VARCHAR(20)
);

CREATE TABLE BOOKCOPIES0033(
No_of_copies INT,
Book_id INT REFERENCES BOOK0033(Book_id)
ON DELETE CASCADE,
Branch_id INT REFERENCES LIBRARYBRANCH0033(Branch_id)
ON DELETE CASCADE,
PRIMARY KEY(Book_id,Branch_id)
);
CREATE TABLE CARD0033(Card_No INT PRIMARY KEY);

CREATE TABLE BOOKLENDING0033(
Dateout DATE,
Duedate DATE,
Book_id INT REFERENCES BOOK0033(Book_id)
ON DELETE CASCADE,
Branch_id INT REFERENCES BOOK0033(Book_id)
ON DELETE CASCADE,
Card_No INT REFERENCES CARD0033(Card_No)
ON DELETE CASCADE,
PRIMARY KEY(Book_id,Branch_id,Card_No)
);
